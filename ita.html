Geometry optimizations and frequency calculations

An instance of geometry optimization of [9]ITAb using MLatom

In[]:
# Create a molecule object and read the geometry from .xyz file as initial guess
_9ITAb = ml.molecule()
_9ITAb.read_from_xyz_file("9ITAb/opt_freq/9ITAb.xyz")

# Here we use UAIQM method
uaiqm = ml.models.methods(method="uaiqm_gfn2xtbstar@cc")

# Optimize the molecule using Gaussian optimizer
mol_opt = ml.optimize_geometry(model=uaiqm, molecule=_9ITAb, program="Gaussian")

# Calculate the frequencies for the optimized molecule
ml.freq(model=uaiqm, molecule=_9ITAb, program="Gaussian")

# Check the frequencies to ensure the structure reach its minima
min(_9ITAb.frequencies)
